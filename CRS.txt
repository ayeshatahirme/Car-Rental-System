
CREATE TABLE CUSTOMER_DETAILS ( 
CNIC CHAR(8) NOT NULL PRIMARY KEY,
FNAME VARCHAR(25) NOT NULL,   
LNAME VARCHAR(25) NOT NULL,   
PHONE_NUMBER INT NOT NULL,   
EMAIL_ID VARCHAR(30) NOT NULL,   
STREET VARCHAR(30) NOT NULL,   
CITY VARCHAR(20) NOT NULL,   
STATE_NAME VARCHAR(20) NOT NULL, 
ZIPCODE INT NOT NULL,   
MEMBERSHIP_TYPE CHAR(1),   
MEMBERSHIP_ID CHAR(5),   
); 

CREATE TABLE CAR_CATEGORY ( 
CATEGORY_NAME VARCHAR(25) NOT NULL PRIMARY KEY,      
COST_PER_DAY INT NOT NULL,   
LATE_FEE_PER_DAY INT NOT NULL,   
); 

CREATE TABLE LOCATION_DETAILS ( 
ZIPCODE INT NOT NULL PRIMARY KEY,
LOCATION_NAME VARCHAR(50) NOT NULL
);

CREATE TABLE CAR ( 
REGISTRATION_NUMBER CHAR(7) NOT NULL PRIMARY KEY,   
MODEL_NAME VARCHAR(25) NOT NULL,   
MAKE VARCHAR(25) NOT NULL,     
AVAILABILITY_FLAG CHAR(1) NOT NULL,  
ZIPCODE INT FOREIGN KEY REFERENCES LOCATION_DETAILS(ZIPCODE),   
CARFK1 VARCHAR(25) FOREIGN KEY REFERENCES CAR_CATEGORY(CATEGORY_NAME)
); 

CREATE TABLE DRIVER_DETAILS (
DRIVER_ID CHAR(8) NOT NULL PRIMARY KEY,
FNAME VARCHAR(25) NOT NULL,
LNAME VARCHAR(25) NOT NULL,
PHONE_NUMBER INT NOT NULL,
/*EMAIL_ID VARCHAR(30) NOT NULL,*/ 
AGE INT NOT NULL,
REG_NO CHAR(7) FOREIGN KEY REFERENCES CAR(REGISTRATION_NUMBER)
);

CREATE TABLE DISCOUNT_DETAILS ( 
DISCOUNT_CODE CHAR(4) NOT NULL PRIMARY KEY,   
DISCOUNT_NAME VARCHAR(25) NOT NULL,   
EXPIRY_DATE DATE NOT NULL,   
DISCOUNT_PERCENTAGE INT  NOT NULL,   
); 

CREATE TABLE RENTAL_CAR_INSURANCE ( 
INSURANCE_CODE CHAR(4) NOT NULL PRIMARY KEY,   
INSURANCE_NAME VARCHAR(50) NOT NULL,   
COVERAGE_TYPE VARCHAR(200) NOT NULL,   
COST_PER_DAY INT NOT NULL,   
); 
 
CREATE TABLE BOOKING_DETAILS ( 
BOOKING_ID CHAR(5) NOT NULL PRIMARY KEY,   
FROM_DATE DATE NOT NULL,   
RET_DATE DATE NOT NULL,   
AMOUNT INT NOT NULL,   
BOOKING_STATUS CHAR(1) NOT NULL,   
PICKUP_LOC  CHAR(4) NOT NULL,   
DROP_LOC CHAR(4) NOT NULL,   
REG_NUM CHAR(7) NOT NULL,   
DL_NUM CHAR(8) NOT NULL,   
INS_CODE CHAR(4),   
ACT_RET_DT_TIME TIMESTAMP,   
DISCOUNT_CODE CHAR(4),   

BOOKINGFK3 CHAR(7) FOREIGN KEY REFERENCES CAR(REGISTRATION_NUMBER),   
BOOKINGFK4 CHAR(8) FOREIGN KEY REFERENCES CUSTOMER_DETAILS(CNIC),   
BOOKINGFK5 CHAR(4) FOREIGN KEY REFERENCES RENTAL_CAR_INSURANCE(INSURANCE_CODE), 
BOOKINGFK6 CHAR(4) FOREIGN KEY REFERENCES DISCOUNT_DETAILS(DISCOUNT_CODE) ); 

CREATE TABLE BILLING_DETAILS ( 
BILL_ID CHAR(6) NOT NULL PRIMARY KEY,   
BILL_DATE DATE NOT NULL,   
BILL_STATUS CHAR(1) NOT NULL,   
DISCOUNT_AMOUNT INT NOT NULL,   
TOTAL_AMOUNT INT NOT NULL,   
TAX_AMOUNT INT NOT NULL,   
BOOKING_ID CHAR(5) NOT NULL,   
TOTAL_LATE_FEE INT NOT NULL,   
); 

CREATE TABLE ADMIN_DETAILS (
ADMIN_ID CHAR(8) NOT NULL PRIMARY KEY,
FNAME VARCHAR(25) NOT NULL,
LNAME VARCHAR(25) NOT NULL,
PHONE_NUMBER INT NOT NULL, 
EMAIL_ID VARCHAR(30) NOT NULL,
);
